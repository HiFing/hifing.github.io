class Solution {
  public int longestConsecutive(int[] nums) {
    Set<Integer> set = new HashSet<>();
    for (int ele : nums)
      set.add(ele);
    int res = 0;
    for (int ele : nums) {
      if (!set.contains(ele - 1)) {
        int count = 1;
        int tmp = ele + 1;
        while (set.contains(tmp)) {
          count++;
          tmp++;
        }
        res = Math.max(res, count);
      }
    }
    return res;
  }
}
