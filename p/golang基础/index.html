type ListNode struct {
  Val  int
  Next *ListNode
}

type NodeHeap []*ListNode

func (nh NodeHeap) Less(i, j int) bool  { return nh[i].Val < nh[j].Val } //å¤§é¡¶å †å°±æ”¹æˆå¤§äºå·
func (nh NodeHeap) Swap(i, j int)       { nh[i], nh[j] = nh[j], nh[i] }

func (nh *NodeHeap) Pop() interface{} {

  res := old[len(old)-1] //å–å‡ºlastOneï¼Œä½†æ˜¯å¯¹äºåˆ‡ç‰‡æ¥è¯´æ˜¯å»é™¤å¤´ä¸€ä¸ªï¼Œå³å †é¡¶å…ƒç´

func mergeKLists(lists []*ListNode) *ListNode {
  h := NodeHeap{}
  heap.Init(&h)
  for i := 0; i < len(lists); i++ {
    head := lists[i]
    for head != nil {
      heap.Push(&h, head)
      head = head.Next
    }
  }
  blank := new(ListNode)
  follower := blank
  for i := 0; i < len(h); i++ {
    fmt.Print(h[i].Val, " ")
  }
  for len(h) > 0 {
    follower.Next = heap.Pop(&h).(*ListNode)
    follower = follower.Next
  }
  follower.Next = nil
  return blank.Next
}
