type ListNode struct {

type NodeHeap []*ListNode

func mergeKLists(lists []*ListNode) *ListNode {
  h := NodeHeap{}
  heap.Init(&h)
  for i := 0; i < len(lists); i++ {
    head := lists[i]
    for head != nil {
      heap.Push(&h, head)
      head = head.Next
    }
  }
  blank := new(ListNode)
  follower := blank
  for i := 0; i < len(h); i++ {
    fmt.Print(h[i].Val, " ")
  }
  for len(h) > 0 {
    follower.Next = heap.Pop(&h).(*ListNode)
    follower = follower.Next
  }
  follower.Next = nil
  return blank.Next
}
