import (
  "container/heap"
  "fmt"
)

type ListNode struct {
  Val  int
  Next *ListNode
}

type NodeHeap []*ListNode

func (nh NodeHeap) Less(i, j int) bool  { return nh[i].Val < nh[j].Val } //Â§ßÈ°∂Â†ÜÂ∞±ÊîπÊàêÂ§ß‰∫éÂè∑
func (nh NodeHeap) Swap(i, j int)       { nh[i], nh[j] = nh[j], nh[i] }
func (nh NodeHeap) Len() int            { return len(nh) }
func (nh *NodeHeap) Push(x interface{}) { *nh = append(*nh, x.(*ListNode)) }

//Ê≥®ÊÑèPopÁöÑÂÜôÊ≥ïÔºÅÔºÅÔºÅ
func (nh *NodeHeap) Pop() interface{} {
  old := *nh
  res := old[len(old)-1] //ÂèñÂá∫lastOneÔºå‰ΩÜÊòØÂØπ‰∫éÂàáÁâáÊù•ËØ¥ÊòØÂéªÈô§Â§¥‰∏Ä‰∏™ÔºåÂç≥Â†ÜÈ°∂ÂÖÉÁ¥†
  *nh = old[:len(old)-1]
  return res
}

func mergeKLists(lists []*ListNode) *ListNode {
  h := NodeHeap{}
  heap.Init(&h)
  for i := 0; i < len(lists); i++ {
    head := lists[i]
    for head != nil {
      heap.Push(&h, head)
      head = head.Next
    }
  }
  blank := new(ListNode)
  follower := blank
  for i := 0; i < len(h); i++ {
    fmt.Print(h[i].Val, " ")
  }
  for len(h) > 0 {
    follower.Next = heap.Pop(&h).(*ListNode)
    follower = follower.Next
  }
  follower.Next = nil
  return blank.Next
}
