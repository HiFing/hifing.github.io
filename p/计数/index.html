class Solution {
  public int majorityElement(int[] nums) {
    int now = nums[0];
    int count = 1;
    for (int i = 1; i < nums.length; i++) {
      if (nums[i] == now)
        count++;
      else {
        if (count > 0)
          count--;
        else {
          count = 1;
          now = nums[i];
        }
      }
    }
    return now;
  }
}
