class Solution {
  public void reverse(int[] nums, int l, int r) {
    int tmp;
    while (l < r) {
      tmp = nums[l];
      nums[l] = nums[r];
      nums[r] = tmp;
      l++;
      r--;
    }
  }

  public void nextPermutation(int[] nums) {
    int i = nums.length - 1;
    for (; i > 0; i--) {
      if (nums[i] > nums[i - 1])
        break;
    }
    if (i == 0)
      reverse(nums, 0, nums.length - 1);
    else {
      for (int j = nums.length - 1; j > i - 1; j--) {
        if (nums[j] > nums[i - 1]) {
          int tmp = nums[j];
          nums[j] = nums[i - 1];
          nums[i - 1] = tmp;
          reverse(nums, i, nums.length - 1);
          return;
        }
      }
    }
  }
}
